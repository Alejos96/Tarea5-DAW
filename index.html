1. Control de Flujo y Manejo de Errores

1.1. if...else

let edad = 18;
if (edad >= 18) {
    console.log("Eres mayor de edad");
} else {
    console.log("Eres menor de edad");
}

1.2. switch

let dia = new Date().getDay();
switch(dia) {
    case 0:
        console.log("Domingo");
        break;
    case 1:
        console.log("Lunes");
        break;
    default:
        console.log("Otro día");
}

1.3. try/catch/throw

try {
    let numero = prompt("Ingresa un número");
    if (isNaN(numero)) {
        throw new Error("No es un número válido");
    }
    console.log("Número válido: " + numero);
} catch (error) {
    console.error("Error: " + error.message);
}

1.4. Objetos Error

function dividir(a, b) {
    if (b === 0) {
        throw new Error("No se puede dividir por cero");
    }
    return a / b;
}

2. Bucles e Iteración

2.1. for

for (let i = 0; i < 5; i++) {
    console.log("Iteración: " + i);
}

2.2. while

let contador = 0;
while (contador < 5) {
    console.log("Contador: " + contador);
    contador++;
}

2.3. do...while

let i = 0;
do {
    console.log("Valor: " + i);
    i++;
} while (i < 5);

3. Funciones

3.1. Definir Funciones

// Función declarativa
function saludar(nombre) {
    return "Hola, " + nombre + "!";
}

// Función expresión
const despedir = function(nombre) {
    return "Adiós, " + nombre + "!";
};

// Función flecha
const sumar = (a, b) => a + b;

3.2. Llamar Funciones

function calcularArea(base, altura) {
    return (base * altura) / 2;
}

let area = calcularArea(10, 5);
console.log("Área del triángulo: " + area);

4. Expresiones y Operadores

4.1. Asignación y Comparación

// Asignación
let x = 10;
x += 5;  // x = 15
x -= 3;  // x = 12

// Comparación
console.log(5 == "5");   // true (comparación de valor)
console.log(5 === "5");  // false (comparación estricta)
console.log(10 > 5);     // true
console.log(10 <= 10);   // true

4.2. Operadores Aritméticos

let a = 10, b = 3;
console.log("Suma: " + (a + b));           // 13
console.log("Resta: " + (a - b));          // 7
console.log("Multiplicación: " + (a * b)); // 30
console.log("División: " + (a / b));       // 3.333...
console.log("Módulo: " + (a % b));         // 1
console.log("Potencia: " + (a ** b));      // 1000

4.3. Operadores Lógicos

let verdadero = true, falso = false;
console.log(verdadero && falso);  // false (AND)
console.log(verdadero || falso);  // true (OR)
console.log(!verdadero);          // false (NOT)
console.log(10 > 5 && 20 < 30);   // true

5. Números y Fechas

5.1. Objeto Number

let num = 123.456;
console.log(num.toFixed(2));        // "123.46"
console.log(num.toPrecision(4));    // "123.5"
console.log(Number.isInteger(10));  // true
console.log(Number.isNaN(NaN));     // true
console.log(Number.MAX_VALUE);      // 1.7976931348623157e+308

5.2. Objeto Math

console.log(Math.PI);              // 3.14159...
console.log(Math.round(4.7));      // 5
console.log(Math.ceil(4.2));       // 5
console.log(Math.floor(4.8));      // 4
console.log(Math.random());        // Número aleatorio entre 0 y 1
console.log(Math.max(1, 5, 3));    // 5
console.log(Math.pow(2, 3));       // 8

5.3. Objeto Date

let fecha = new Date();
console.log(fecha.toString());
console.log(fecha.getFullYear());
console.log(fecha.getMonth() + 1);
console.log(fecha.getDate());
console.log(fecha.getHours() + ":" + fecha.getMinutes());

6. Formateo de Texto

6.1. Cadenas Literales

let nombre = "Juan";
let edad = 25;

// Template literals
let mensaje = `Hola, mi nombre es ${nombre} y tengo ${edad} años`;

// Cadenas multilínea
let parrafo = `Esta es una línea
Esta es otra línea
Y esta es la tercera línea`;

6.2. Objeto String

let texto = "JavaScript es genial";
console.log(texto.length);              // 20
console.log(texto.toUpperCase());       // "JAVASCRIPT ES GENIAL"
console.log(texto.toLowerCase());       // "javascript es genial"
console.log(texto.substring(0, 10));    // "JavaScript"
console.log(texto.split(" "));          // ["JavaScript", "es", "genial"]
console.log(texto.includes("genial"));  // true
console.log(texto.replace("genial", "increíble"));

7. Colecciones Indexadas

7.1. Arreglos

let frutas = ["manzana", "banana", "naranja"];
console.log(frutas[0]);           // "manzana"
frutas.push("uva");               // Agregar al final
frutas.pop();                     // Eliminar del final
frutas.unshift("fresa");          // Agregar al inicio
console.log(frutas.length);       // 4

// Métodos de arrays
let numeros = [1, 2, 3, 4, 5];
let dobles = numeros.map(n => n * 2);
let pares = numeros.filter(n => n % 2 === 0);
let suma = numeros.reduce((acc, n) => acc + n, 0);

7.2. Arreglos Tipados

// Int8Array: enteros de 8 bits con signo (-128 a 127)
let int8 = new Int8Array([1, 2, 3, 4, 5]);

// Uint8Array: enteros de 8 bits sin signo (0 a 255)
let uint8 = new Uint8Array([10, 20, 30]);

// Float32Array: números de punto flotante de 32 bits
let float32 = new Float32Array([1.5, 2.5, 3.5]);

console.log(int8);
console.log(uint8);
console.log(float32);
